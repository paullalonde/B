584 template <class VIEW> VIEW*
585 PredefinedView::SetupTypedView(
586  const HIViewID& inViewID,
587  HIViewRef inViewRef,
588  HIViewRef inSuperview,
589  const Rect* inFrame,
590  Nib* inFromNib,
591  const ViewFactory* inFactory)
592 {
593  View* view;
594  VIEW* typedView;
595 
596  view = SetupView(inViewID, inViewRef, inSuperview, inFrame, inFromNib, inFactory);
597  typedView = dynamic_cast<VIEW*>(view);
598 
599  // If we hit this assert, it means that the type of the view doesn't match the VIEW
600  // template parameter. This is a logic error and should never happen in the course
601  // of normal program execution.
602 
603  B_ASSERT(typedView != NULL);
604 
605  // Throw instead of continuing.
606  if (typedView == NULL)
607  {
608  // Release the view. This will usually free the View object (since there is
609  // normally only one reference to the view at this point).
610  CFRelease(inViewRef);
611 
612  B_THROW(std::logic_error("PredefinedView::SetupTypedView type mismatch"));
613  }
614 
615  return (typedView);
616 }
