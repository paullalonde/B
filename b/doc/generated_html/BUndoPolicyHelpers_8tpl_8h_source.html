62 template <class UNDO_POLICY> void
63 UndoPolicyEvents<UNDO_POLICY>::RegisterAppleEvents(AEObjectSupport& ioObjectSupport)
64 {
65  // Define the Undo event if it isn't already defined. This way apps don't have to
66  // bother with defining the event in their sdef file.
67 
68  if (!ioObjectSupport.IsEventDefined(kAEMiscStandards, kAEUndo))
69  {
70  ioObjectSupport.DefineEvent(kAEMiscStandards, kAEUndo,
71  AEInfo::kEventDOIllegal,
72  AEInfo::kEventResultActionNone);
73  }
74 
75  // Ditto for Redo.
76 
77  if (!ioObjectSupport.IsEventDefined(kAEMiscStandards, kAERedo))
78  {
79  ioObjectSupport.DefineEvent(kAEMiscStandards, kAERedo,
80  AEInfo::kEventDOIllegal,
81  AEInfo::kEventResultActionNone);
82  }
83 
84  ioObjectSupport.SetClassEventHandler(cApplication, this, &ImplType::HandleUndoAppleEvent);
85  ioObjectSupport.SetClassEventHandler(cApplication, this, &ImplType::HandleRedoAppleEvent);
86 }

120 template <class UNDO_POLICY> bool
121 UndoPolicyEvents<UNDO_POLICY>::CommandProcess(
122  Event<kEventClassCommand, kEventCommandProcess>& event)
123 {
124  bool handled = true;
125 
126  switch (event.mHICommand.commandID)
127  {
128  case kHICommandUndo:
129  mPolicy.Undo();
130  break;
131 
132  case kHICommandRedo:
133  mPolicy.Redo();
134  break;
135 
136  default:
137  handled = false;
138  break;
139  }
140 
141  return (handled);
142 }

145 template <class UNDO_POLICY> bool
146 UndoPolicyEvents<UNDO_POLICY>::CommandUpdateStatus(
147  Event<kEventClassCommand, kEventCommandUpdateStatus>& event)
148 {
149  bool handled = true;
150 
151  switch (event.mHICommand.commandID)
152  {
153  case kHICommandUndo:
154  event.mData.SetEnabled(mPolicy.CanUndo());
155  event.mData.SetText(mPolicy.GetUndoText());
156  break;
157 
158  case kHICommandRedo:
159  event.mData.SetEnabled(mPolicy.CanRedo());
160  event.mData.SetText(mPolicy.GetRedoText());
161  break;
162 
163  default:
164  handled = false;
165  break;
166  }
167 
168  return (handled);
169 }
