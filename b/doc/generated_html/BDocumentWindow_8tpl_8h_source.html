136 template <class DOCUMENT> void
137 DocumentWindow<DOCUMENT>::LoadProxyIcon()
138 {
139  Url url = mDocument->GetUrl();
140 
141  if (url.Empty())
142  {
143  // The document has never been saved, so it shouldn't have a proxy icon.
144 
145  ClearProxy();
146  }
147  else if (url.IsFileUrl())
148  {
149  // If the document has a file URL, use it as the proxy instead of the icon
150  // directly. This allows the Window Manager to extract the icon itself.
151 
152  SetProxy(url);
153  }
154  else
155  {
156  SetProxy(mDocument->GetIcon());
157  }
158 }

266 template <class DOCUMENT> bool
267 DocumentWindow<DOCUMENT>::RouteEventToDocument(EventBase& event)
268 {
269  OSStatus err;
270 
271  EventParam<kEventParamWindowRef>::Set(event, *this);
272 
273  err = event.Send(mDocument->GetEventTarget(), std::nothrow);
274 
275  if ((err != noErr) && (err != eventNotHandledErr))
276  B_THROW_STATUS(err);
277 
278  return (err == noErr);
279 }
