236 template <class DOC_POLICY, class UNDO_POLICY, class PRINT_POLICY> void
237 Application<DOC_POLICY, UNDO_POLICY, PRINT_POLICY>::RegisterAppleEvents(
238  AEObjectSupport& ioObjectSupport)
239 {
240  // Install event handlers for events lacking a direct parameter.
241 
242  // The Open Application event is required.
243  B_ASSERT(ioObjectSupport.IsEventDefined(kCoreEventClass, kAEOpenApplication));
244  ioObjectSupport.SetClassEventHandler(this,
245  &Application<DOC_POLICY, UNDO_POLICY, PRINT_POLICY>::HandleOpenApplicationAppleEvent);
246 
247  // The Reopen Application event is required.
248  B_ASSERT(ioObjectSupport.IsEventDefined(kCoreEventClass, kAEReopenApplication));
249  ioObjectSupport.SetClassEventHandler(this,
250  &Application<DOC_POLICY, UNDO_POLICY, PRINT_POLICY>::HandleReopenApplicationAppleEvent);
251 
252  // The Quit Application event is required.
253  B_ASSERT(ioObjectSupport.IsEventDefined(kCoreEventClass, kAEQuitApplication));
254  ioObjectSupport.SetClassEventHandler(this,
255  &Application<DOC_POLICY, UNDO_POLICY, PRINT_POLICY>::HandleQuitApplicationAppleEvent);
256 
257  // The Show Preferences event isn't required.
258  if (ioObjectSupport.IsEventDefined(kCoreEventClass, kAEShowPreferences))
259  {
260  ioObjectSupport.SetClassEventHandler(this,
261  &Application<DOC_POLICY, UNDO_POLICY, PRINT_POLICY>::HandleShowPreferencesAppleEvent);
262  }
263 
264  // Give the document policy a chance to register AppleEvents for itself.
265  mDocumentPolicy.RegisterAppleEvents(ioObjectSupport);
266 
267  // Ditto for the Undo policy.
268  mUndoPolicy.RegisterAppleEvents(ioObjectSupport);
269 }

298 template <class DOC_POLICY, class UNDO_POLICY, class PRINT_POLICY> void
299 Application<DOC_POLICY, UNDO_POLICY, PRINT_POLICY>::PostprocessMenuBar()
300 {
301  OSPtr<MenuRef> rootMenu(AcquireRootMenu(), from_copy);
302  std::vector<MenuRef> appMenus;
303  OSPtr<CFSetRef> appMenusSet;
304  MenuRef applicationMenu, windowMenu, helpMenu;
305  OSStatus err;
306 
307  // We consider three menus to be "system" menus: Application, Window, and Help.
308  //
309  // How do we identify a particular menu? By searching for commands that reside
310  // in the menu...
311 
312  // The Application menu contains the "About" command.
313  err = GetIndMenuItemWithCommandID(NULL, kHICommandAbout, 1, &applicationMenu, NULL);
314  if (err != noErr)
315  applicationMenu = NULL;
316 
317  // The Window menu contains special markers for the window list.
318  err = GetIndMenuItemWithCommandID(NULL, kHICommandWindowListSeparator, 1, &windowMenu, NULL);
319  if (err != noErr)
320  windowMenu = NULL;
321 
322  // The Help menu is obtained through a dedicated API.
323  err = HMGetHelpMenu(&helpMenu, NULL);
324  if (err != noErr)
325  helpMenu = NULL;
326 
327  MenuItemIndex count = CountMenuItems(rootMenu);
328 
329  for (MenuItemIndex i = 1; i <= count; i++)
330  {
331  MenuRef menu;
332 
333  err = GetMenuItemHierarchicalMenu(rootMenu, i, &menu);
334 
335  // Skip if this isn't a menu.
336  if ((err != noErr) || (menu == NULL))
337  continue;
338 
339  // Skip if this is a system menu.
340  if ((menu == applicationMenu) || (menu == windowMenu) || (menu == helpMenu))
341  continue;
342 
343  appMenus.push_back(menu);
344  AddSubmenusToVector(menu, appMenus);
345  }
346 
347  if (!appMenus.empty())
348  {
349  appMenusSet.reset(CFSetCreate(NULL,
350  (const void **) &appMenus[0],
351  appMenus.size(),
352  &kCFTypeSetCallBacks),
353  from_copy);
354  }
355 
356  mAppMenus = appMenusSet;
357 }

360 template <class DOC_POLICY, class UNDO_POLICY, class PRINT_POLICY> void
361 Application<DOC_POLICY, UNDO_POLICY, PRINT_POLICY>::AddSubmenusToVector(
362  MenuRef inMenu,
363  std::vector<MenuRef>& ioVector)
364 {
365  MenuItemIndex count = CountMenuItems(inMenu);
366 
367  for (MenuItemIndex i = 1; i <= count; i++)
368  {
369  MenuRef subMenu;
370  OSStatus err;
371 
372  err = GetMenuItemHierarchicalMenu(inMenu, i, &subMenu);
373 
374  if ((err == noErr) && (subMenu != NULL))
375  {
376  ioVector.push_back(subMenu);
377  AddSubmenusToVector(subMenu, ioVector);
378  }
379  }
380 }

784 template <class DOC_POLICY, class UNDO_POLICY, class PRINT_POLICY> void
785 Application<DOC_POLICY, UNDO_POLICY, PRINT_POLICY>::WriteProperty(
786  DescType inPropertyID,
787  AEWriter& ioWriter) const
788 {
789  switch (inPropertyID)
790  {
791  case pBestType:
792  ioWriter.template Write<typeType>(GetClassID());
793  break;
794 
795  case pName:
796  ioWriter.template Write<typeUTF16ExternalRepresentation>(GetName());
797  break;
798 
799  case pVersion:
800  ioWriter.template Write<typeUTF16ExternalRepresentation>(mBundle.ShortVersionString());
801  break;
802 
803  case pIsFrontProcess:
804  ioWriter.template Write<typeBoolean>(MoreProcIsProcessAtFront(NULL));
805  break;
806 
807  default:
