708 template <DescType TYPE>
709 bool
710 AEObjectSupport::CompareNumericData(
711  DescType inOperator,
712  typename DescParam<TYPE>::ValueType inValue1,
713  typename DescParam<TYPE>::ValueType inValue2)
714 {
715  // Numeric stuff. We support all the standard relational operators.
716 
717  bool result = false;
718 
719  switch (inOperator)
720  {
721  case kAEEquals: result = (inValue1 == inValue2); break;
722  case kAEGreaterThanEquals: result = (inValue1 >= inValue2); break;
723  case kAEGreaterThan: result = (inValue1 > inValue2); break;
724  case kAELessThan: result = (inValue1 < inValue2); break;
725  case kAELessThanEquals: result = (inValue1 <= inValue2); break;
726 
727  case kAEBeginsWith:
728  case kAEEndsWith:
729  case kAEContains:
730  B_THROW(AECantRelateObjectsException());
731  break;
732 
733  default:
734  B_THROW(AEUnrecognisedOperatorException());
735  break;
736  }
737 
738  return result;
739 }
