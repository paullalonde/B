253 template <class NAV_DIALOG> void
254 NavDialog::Enter(
255  std::auto_ptr<NAV_DIALOG> inDialogPtr)
256 {
257  boost::function_requires< boost::ConvertibleConcept<NAV_DIALOG*, B::NavDialog*> >();
258 
259  // The caller has given us control over the lifetime of the dialog, so dispose of it
260  // upon dismissal.
261  inDialogPtr->SetDeleteOnDismiss(true);
262 
263  // Enter the dialog. This may or may not return until the dialog has been dismissed.
264  inDialogPtr->Enter();
265 
266  // We don't need this anymore.
267  inDialogPtr.release();
