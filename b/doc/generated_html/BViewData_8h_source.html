144 // ------------------------------------------------------------------------------------------
148 template <typename T> inline OSStatus
149 ViewDataTrait<T>::Get(
150  HIViewRef view,
151  ControlPartCode part,
152  ResType tag,
153  T& value,
154  const std::nothrow_t&)
155 {
156  ::Size size;
157  OSStatus err;
158 
159  err = GetControlData(view, part, tag, sizeof(value), &value, &size);
160  if ((err == noErr) && (size != sizeof(value)))
161  err = errDataSizeMismatch;
162 
163  return (err);
164 }
165 
166 // ------------------------------------------------------------------------------------------
167 template <typename T> void
168 ViewDataTrait<T>::Get(
169  HIViewRef view,
170  ControlPartCode part,
171  ResType tag,
172  T& value)
173 {
174  OSStatus err;
175 
176  err = Get(view, part, tag, value, std::nothrow);
177  B_THROW_IF_STATUS(err);
178 }

215 template <typename T> OSStatus
216 ViewDataTrait<T>::Get(
217  const void* data,
218  size_t size,
219  T& value)
220 {
221  OSStatus err = noErr;
222 
223  if (size == sizeof(value))
224  value = * reinterpret_cast<const T*>(data);
225  else
226  err = errDataSizeMismatch;
227 
228  return (err);
229 }

231 // ------------------------------------------------------------------------------------------
237 template <typename T> OSStatus
238 ViewDataTrait<T>::Set(
239  void* data,
240  size_t size,
241  const T& value)
242 {
243  OSStatus err = noErr;
244 
245  if (size == sizeof(value))
246  * reinterpret_cast<T*>(data) = value;
247  else
248  err = errDataSizeMismatch;
249 
250  return (err);
251 }
