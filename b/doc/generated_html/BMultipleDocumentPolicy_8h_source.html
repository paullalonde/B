312 template <class DOC_FACTORY> template <class PRED> AbstractDocument*
313 MultipleDocumentPolicy<DOC_FACTORY>::FindDocumentMatching(PRED pred) const
314 {
315  AbstractDocument* document = NULL;
316  DocumentConstIterator it;
317 
318  it = std::find_if(mDocuments.begin(), mDocuments.end(), pred);
319 
320  if (it != mDocuments.end())
321  document = it->second.get();
322 
323  return (document);
324 }
