389 template <typename T> void
390 MutableArray<T>::init_array(CFArrayRef cfarr, const T* values, size_t num, CFAllocatorRef allocator)
391 {
392  CFMutableArrayRef mutRef;
393 
394  if (cfarr != NULL)
395  {
396  mutRef = CFArrayCreateMutableCopy(allocator, 0, cfarr);
397  }
398  else
399  {
400  mutRef = CFArrayCreateMutable(allocator, 0, &kCFTypeArrayCallBacks);
401 
402  if ((mutRef != NULL) && (num > 0))
403  {
404  CFArrayReplaceValues(mutRef, ::CFRangeMake(0, 0), (const void **) values, num);
405  }
406  }
407 
408  B_THROW_IF_NULL(mutRef);
409 
410  mRef = mutRef;
411 }
